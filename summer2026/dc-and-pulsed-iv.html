Measurement Results Two methods were used to obtain the IV characteristics of a SKY65050 transistor: DC IV and Pulsed IV.
A Keysight SMU was used for the DC measurement and an AMCAD SMU was used for the Pulsed measurement.
Deviations from the expected curves could be attributed to trapping effects in the GaAs.
Pulsed measurements result in less time for the device to overheat during testing.

BLOCK_SIZE = 54  # lines per curve


def parse_iv_csv(filepath):
    with open(filepath) as fh:
        lines = fh.readlines()

    curves = []
    for start in range(0, len(lines), BLOCK_SIZE):
        block = lines[start : start + BLOCK_SIZE]
        if not block:
            break
        label_match = re.search(r"Input\s*=\s*([\-\d.]+\s*V)", block[0])
        label = label_match.group(1) if label_match else f"Curve {len(curves)}"
        text = "".join(block[1:])
        df = pd.read_csv(StringIO(text), header=0)
        df = df.apply(pd.to_numeric, errors="coerce")
        curves.append((label, df))

    return curves


def plot_single(filepath, title, ax_x_col, ax_y_col, xlabel, ylabel,
                cmap="autumn"):
    curves = parse_iv_csv(filepath)
    n = len(curves)
    colors = plt.get_cmap(cmap)(np.linspace(0.15, 0.85, n))

    fig, ax = plt.subplots(figsize=(8, 5))
    for (label, df), color in zip(curves, colors):
        ax.plot(df[ax_x_col], df[ax_y_col],
                linewidth=1.5,
                label=label, color=color)

    ax.set_title(title)
    ax.set_xlabel(xlabel)
    ax.set_ylabel(ylabel)
    ax.legend(title="Input bias", fontsize=8, title_fontsize=8)
    ax.grid(True, linestyle="--", alpha=0.4)
    fig.tight_layout()
    return fig


def plot_comparison(filepath1, filepath2, title, ax_x_col, ax_y_col,
                    xlabel, ylabel,
                    label1=None, label2=None,
                    color1="red", color2="blue"):
    label1 = label1 or filepath1
    label2 = label2 or filepath2

    curves1 = parse_iv_csv(filepath1)
    curves2 = parse_iv_csv(filepath2)

    fig, ax = plt.subplots(figsize=(8, 5))

    for i, (curve_label, df) in enumerate(curves1):
        ax.plot(df[ax_x_col], df[ax_y_col],
                linewidth=1.5, color=color1,
                label=label1 if i == 0 else "_nolegend_")

    for i, (curve_label, df) in enumerate(curves2):
        ax.plot(df[ax_x_col], df[ax_y_col],
                linewidth=1.5, color=color2,
                label=label2 if i == 0 else "_nolegend_")

    ax.set_title(title)
    ax.set_xlabel(xlabel)
    ax.set_ylabel(ylabel)
    ax.legend(fontsize=9)
    ax.grid(True, linestyle="--", alpha=0.4)
    fig.tight_layout()
    return fig
